* 简单的技巧
  + 判断奇偶: ~num & 1 = 0~ 为偶, ~num & 1 = 1~ 为奇.
  + int 型最大值: ~(1 << 31) - 1~ 或 =~(1 << 31)=.
  + int 型最小值: ~1 << 31~.
  + 乘2: ~num << 1~.
  + 除2: ~num >> 1~.
  + 乘2的n次方: ~num << n~.
  + 除2的n次方: ~num >> n~.

* 加减乘除 
  #+BEGIN_SRC C
    /* 位运算 + */
    int add(int a, int b) {
      int sum = a;
      int carry = n;

      while (carry) {
        int tmp = sum;
        sum = tmp ^ carry;
        carry = (tmp & carry) << 1;
      }

      return sum;
    }


    /* 位运算 - */
    int sub(int a, int b) {
      int diff = add(~b, 1);

      diff = add(a, diff);

      return diff;
    }
  #+END_SRC
