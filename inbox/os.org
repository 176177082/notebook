* 堆栈
  + 堆向上生长， 栈向下生长
    #+BEGIN_SRC C
      #include <stdio.h>
      #include <stdlib.h>

      int main(int ragc, char* argv[]) {
        int a = 10;
        int b = 10;
        int* heap = (int*)malloc(10 * sizeof(int));

        printf("a: %p, b: %p\n", &a, &b);
        for (int i = 0; i < 10; ++i) {
          printf("heap[%d]: %p\n", i, &heap[i]);
        }

        return 0;
      }
    #+END_SRC
