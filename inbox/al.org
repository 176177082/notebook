    def BFS(root):
        queue, result = deque(), []

        queue.append(root)

        while len(queue) > 0:
            node = queue.popleft()

            result.append(node.val)

            if node.left:
                queue.append(node.left)

            if node.right:
                queue.append(node.right)

        return result
