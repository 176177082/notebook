    public static <T> T convert(Object array, Class<T> clazz) {
      Object newArray = Array.newInstance(clazz, Array.getLength(array));
      for (int i = 0; i < Array.getLength(array); ++i) {
        Array.set(newArray, i, Array.get(array, i));
